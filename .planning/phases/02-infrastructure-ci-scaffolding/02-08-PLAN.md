quality-gates.yml structure:

name: Phase 2 quality gates

on:
  deployment_status:        # CF Pages emits after each preview deploy
  workflow_dispatch:        # manual trigger for debugging

jobs:
  # Skip non-success deployments + non-CF environments
  preflight:
    if: github.event.deployment_status.state == 'success'
    runs-on: ubuntu-latest
    outputs:
      preview_url: ${{ steps.url.outputs.preview_url }}
    steps:
      - id: url
        run: echo "preview_url=${{ github.event.deployment_status.environment_url }}" >> $GITHUB_OUTPUT

  visual-regression:
    needs: preflight
    runs-on: ubuntu-latest
    steps: [checkout, setup-node, pnpm install, playwright install chromium, pnpm exec playwright test visual-regression.spec.ts]
    env: { PREVIEW_URL: ${{ needs.preflight.outputs.preview_url }} }

  fidelity:
    needs: preflight
    runs-on: ubuntu-latest
    steps: [checkout, setup-python, python3 scripts/verify-fidelity.py --base-url $PREVIEW_URL --color]

  tone-colours:
    needs: preflight
    runs-on: ubuntu-latest
    steps: [checkout, setup-node, pnpm install, playwright install chromium, pnpm exec playwright test tone-colours.spec.ts]
    env: { PREVIEW_URL: ${{ needs.preflight.outputs.preview_url }} }

  js-off:
    needs: preflight
    runs-on: ubuntu-latest
    steps: [checkout, setup-node, pnpm install, playwright install chromium, pnpm exec playwright test js-off.spec.ts]
    env: { PREVIEW_URL: ${{ needs.preflight.outputs.preview_url }} }

  lighthouse:
    needs: preflight
    runs-on: ubuntu-latest
    continue-on-error: true       # SOFT per D-28
    steps:
      - sed substitution of __PREVIEW_URL__ in .lighthouserc.cf.json
      - pnpm exec lhci autorun --config=/tmp/lhci.json

    needs: preflight
    runs-on: ubuntu-latest
    steps:
      - python3 scripts/build-redirects.py --check     # drift gate
      - bash scripts/verify-redirects.sh $PREVIEW_URL  # curl harness

      # CF Pages _redirects verification harness — Phase 2 INF-02
      # Iterates every canonical route in URL-CONTRACT.txt and asserts

      # build-redirects.py promised (200 for canonical paths, 301 for

      - `MAX_FAILS=10` (CI early-bail; print first 10 failures then summary — avoids 100s of failures spam)
    - CLI:
      - `$1` = preview URL (required); usage error if missing
      - Optional `--strict` flag (default lenient: only assert each canonical route is 2xx — Phase 2 doesn't enforce the 301 unslashed-form leg yet because some routes from URL-CONTRACT.txt may not have matching unslashed routes on the preview if CF Pages parser dropped them)

      - `check_url(url, expected_status)`:
        - `actual=$(curl -sI -o /dev/null -w '%{http_code}' --max-time 15 -A "$UA" "$url")`
        - If `[ "$actual" = "$expected_status" ]`: `echo "  [ok]    $url → $actual"`; return 0
        - Else: `echo "  [FAIL]  $url → got $actual, expected $expected_status"`; return 1
      - `main`:
        - Read URL-CONTRACT.txt skipping `#` header lines and blank lines
        - For each route: strip `#fragment`; dedupe (use shell associative array or pipe through `sort -u`)
        - For each canonical route: call `check_url "${preview_url}${route}" 200`
        - Tally pass/fail; print summary; exit 0 if zero fails, 1 otherwise
    - Make executable: `chmod +x scripts/verify-redirects.sh`

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GH Actions runner ↔ deployment_status event | Webhook signed by GitHub; CF Pages GitHub App is the sender. Event payload trusted within the action runtime. |
| GH Actions runner ↔ CF Pages preview origin | Read-only HTTP traffic; same trust surface as any user. |
| GH Actions runner ↔ committed scripts | Runner pulls scripts from the PR's commit; PR review is the human gate. |
| Workflow ↔ existing CI workflows | quality-gates.yml is sibling; existing workflows (D-32) unchanged. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-28 | Spoofing | Forged deployment_status event | accept | GitHub Actions only accepts deployment_status events from authenticated senders (CF Pages GH App). No public path to inject. |
