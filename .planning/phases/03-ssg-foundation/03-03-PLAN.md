       ```
       def render_card_html(row, idx):
           e = lambda v: html.escape(v or '', quote=True)
           row_id = f"r{idx + 1:03d}"  # 1-based, 3-digit (matches data-id="r051" pattern in checker B1)
           slug = _slugify(row.get('Title', ''))
           url = row.get('PDF | Image Link', '') or ''
           thumb = row.get('Modal Image', '') or ''
           alt = row.get('Image Alt Text', '') or row.get('Title', '')
           title = row.get('Title', '')
           desc = row.get('Description Blurb', '')
           agency = row.get('Agency', '')
           date = row.get('Incident Date', '')
           rtype = row.get('Type', '')
           dvids = row.get('DVIDS Video ID', '')

           parts = []
           parts.append(
               f'<article class="arch-card" id="card-{e(slug)}" data-id="{e(row_id)}" '
               f'data-idx="{idx}" data-action="open" data-type="{e(rtype)}" '
               f'data-agency="{e(agency)}" data-date="{e(date)}">'
           )
           if thumb:
               parts.append(
                   f'<img loading="lazy" src="{e(thumb)}" data-fallback="{e(url)}" '
                   f'alt="{e(alt)}" onerror="this.onerror=null;this.src=this.dataset.fallback||\'\'" />'
               )
           parts.append(f'<h3 class="card-title">{e(title)}</h3>')
           if desc:
               parts.append(f'<p class="card-desc">{e(desc)}</p>')
           parts.append('<dl class="card-meta">')
           parts.append(f'<dt>Agency</dt><dd>{e(agency)}</dd>')
           parts.append(f'<dt>Date</dt><dd>{e(date)}</dd>')
           parts.append(f'<dt>Type</dt><dd>{e(rtype)}</dd>')
           parts.append('</dl>')
           parts.append('<div class="card-actions">')
           if url:
               parts.append(f'<a href="{e(url)}" class="btn-open" data-action="open" data-idx="{idx}">Open</a>')
               parts.append(f'<a href="{e(url)}" class="btn-download" download>Download</a>')
           parts.append(f'<a href="{SOURCE_URL}" class="btn-source" target="_blank" rel="noopener">Source ↗</a>')
           if dvids:

                   f'<a href="https://www.dvidshub.net/video/{e(dvids)}" class="btn-dvids" '
                   f'target="_blank" rel="noopener">DVIDS ↗</a>'
               )
           parts.append('</div>')
           parts.append('</article>')
           return ''.join(parts)
       ```
